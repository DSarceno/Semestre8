\begin{ejercicio}
	Teniendo la forma de cada operador y sabiendo que $[x,p] = i\hbar$, entonces
	$$
		[a,a^\dagger] = \frac{m\omega}{2\hbar} \qty[ x + \frac{ip}{m\omega}, x - \frac{ip}{m\omega} ] = \frac{m\omega}{2\hbar} \qty(-\frac{2i}{m\omega} [x,p]) = \mathbb{1} .
	$$
\end{ejercicio}

		$$
\end{ejercicio}
